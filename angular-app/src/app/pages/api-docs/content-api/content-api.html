paginate all results 
async function fetchAllQuotes(tag) {
  const quotes = [];
  let page = 1;
  let totalPages = 1;

  while (page <= totalPages) {
    const res = await fetch(
      `https://api.quotable.io/quotes?tags=${tag}&page=${page}&limit=150`
    );
    const data = await res.json();

    quotes.push(...data.results);
    totalPages = data.totalPages;
    page++;
  }

  return quotes; // full dataset, de-paginated
}
