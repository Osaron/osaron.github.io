const getQuote = async (tags = '', maxLength = '') => {
  const params = new URLSearchParams();
  if (tags)      params.set('tags',      tags);
  if (maxLength) params.set('maxLength', maxLength);

  const url = `https://api.quotable.io/quotes/random?${params}`;
  const [quote] = await fetch(url).then(r => r.json());
  return quote;
};

const q = await getQuote('wisdom', '120');
console.log(`"${q.content}" — ${q.author}`);

requests 
import requests

def get_quote(tags='', max_length=None):
    params = {}
    if tags:       params['tags']      = tags
    if max_length: params['maxLength'] = max_length

    r = requests.get(
        'https://api.quotable.io/quotes/random',
        params=params
    )
    r.raise_for_status()
    return r.json()[0]

q = get_quote(tags='wisdom', max_length=120)
print(f'"{q["content"]}" — {q["author"]}')
