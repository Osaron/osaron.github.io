import React, { useState, useEffect } from 'react';
function ContentList() {
  const [content, setContent] = useState([]);
  useEffect(() => {
    client.content.list({ type: 'blog', status: 'published', limit: 10 })
      .then(r => setContent(r.data));
  }, []);
  return content.map(item => <article key={item.id}><h2>{item.title}</h2></article>);
}
export default {
  data() { return { content: [] }; },
  async mounted() {
    const r = await client.content.list({ type: 'blog', status: 'published' });
    this.content = r.data;
  }
};

class RealTimeManager {
  constructor(apiKey) {
    this.ws = new WebSocket(`wss://api.oscarrondon.com/v1/ws?token=${apiKey}`);
    this.ws.onmessage = (e) => this.handleMessage(JSON.parse(e.data));
  }
  handleMessage(msg) {
    if (msg.type === 'content.updated') this.updateUI(msg.data);
  }
}
